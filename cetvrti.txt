    public static <T> Map<String, Boolean> validatePattern(
            List<T> list,
            List<T> pattern) {

        AtomicInteger p = new AtomicInteger();
        
        //PROVERA DALI JE OBRAZAC ISPRAVAN BEZ OBZIRA NA BROJ DUPLKATATA
        boolean patternMatches = list.stream().allMatch(item -> {
            int i = p.get();

            if (item.equals(pattern.get(i))) {
                return true;
            }

            if (i < pattern.size() - 1
                    && item.equals(pattern.get(i + 1))) {
                p.incrementAndGet();
                return true;
            }

            return false;
        });
        //PROVERAVA DALI LISTA BEZ DUPLIKATA PRATI OBRAZAC
        boolean sameDistinctValues = list.stream()
                .distinct()
                .collect(Collectors.toList())
                .equals(pattern);

        Map<String, Boolean> result = new HashMap<>();
        result.put("patternMatches", patternMatches);
        result.put("sameDistinctValues", sameDistinctValues);

        return result;
    }
