package estam.estamPaket;

import java.util.Arrays;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.concurrent.atomic.AtomicInteger;
import java.util.stream.Collectors;

public class EstamClassDva {

    public static void main(String[] args) {

        List<String> list = Arrays.asList("eki","zef","gzim","gani","gani","saki");
        List<String> ptrn = Arrays.asList("eki","zef","gzim","gani");

        Map<String, Boolean> results = validatePattern(list, ptrn);
        
        System.out.println(results);
        
    }
    
    
    
    
    
    public static <T> Map<String, Boolean> validatePattern(
            List<T> list,
            List<T> pattern) {

        AtomicInteger p = new AtomicInteger();

        boolean patternMatches = list.stream().allMatch(item -> {
            int i = p.get();

            if (item.equals(pattern.get(i))) {
                return true;
            }

            if (i < pattern.size() - 1
                    && item.equals(pattern.get(i + 1))) {
                p.incrementAndGet();
                return true;
            }

            return false;
        });

        boolean sameDistinctValues = list.stream()
                .distinct()
                .collect(Collectors.toList())
                .equals(pattern);

        Map<String, Boolean> result = new HashMap<>();
        result.put("patternMatches", patternMatches);
        result.put("sameDistinctValues", sameDistinctValues);

        return result;
    }
    
    
    
    
    
    
    
    
    
    
    
    
    
}
